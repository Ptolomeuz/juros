<script>

const principal = 100;

const taxa = 0.001; // juros por minuto

const startTime = new Date();


function atualizar() {

const now = new Date();

const minutosPassados = (now - startTime) / 60000; // ms → minutos

const valorAtual = principal * Math.pow(1 + taxa, minutosPassados);


document.getElementById('valor').textContent = "R$ " + valorAtual.toFixed(2);

document.getElementById('tempo').textContent = "⏱ Tempo desde o empréstimo: " + Math.floor(minutosPassados) + " minutos";

}


setInterval(atualizar, 1000); // atualiza a cada 1 segundo

</script>
